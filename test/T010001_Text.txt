  ༄༅།  ། གཉིས་པ་ཕར་ཕྱིན་ཐ་མ་གཉིས་ལ་སློབ་ཚུལ་འཆད་པའི་སྒོ་ནས་མཆོད་པར་བརྗོད་པའི་ཚུལ་ནི།  རྗེ་བཙུན་དམ་པ་ཐུགས་རྗེ་ཆེན་པོ་དང་ལྡན་པ་རྣམས་ཀྱི་ཞབས་ལ་གུས་པས་ཕྱག་འཚལ་ལོ། ། ཞེས་པ་སྟེ་ཆོས་འདི་གང་ལས་བརྒྱུད་པའི་རྗེ་བླ་མ་རང་ཉིད་ཀྱི་དངོས་བརྒྱུད་ཀྱི་བླ་མ་རྣམས་ཏེ། དེ་ཡང་སྔར་བཤད་པ་ལྟར་ཐུབ་པའི་དབང་པོ་ཉིད་ནས་བྱམས་པ་ཐོགས་མེད་སོགས་རྒྱ་ཆེན་སྤྱོད་བརྒྱུད་དང་།  འཇམ་དབྱངས་ཀླུ་སྒྲུབ་ཞི་བ་ལྷ་སོགས་ཟབ་མོ་ལྟ་བརྒྱུད་ཀྱི་བླ་མ་རྣམས་དང་།  ཁྱད་པར་རྗེ་ཉིད་ཀྱི་ཞལ་ནས།  བླ་མའི་དྲིན་གྱིས་མཐོང་བའི་ཆེ།  །བདག་གི་ཡིད་ཀྱིས་ངལ་བསོ་ཐོབ།  །ཅེས་གསུངས་པ་ལྟར་དོན་ལ་རྗེ་བླ་མ་ཉིད་མགོན་པོ་འཇམ་དབྱངས་དང་གཉིས་སུ་མ་མཆིས་ཀྱང་གདུལ་བྱ་ཐུན་མོང་བའི་ངོར་ཟབ་མོ་དབུ་མའི་ལྟ་བ་འཇམ་དཔལ་དབྱངས་ལས་དངོས་སུ་གསན་པའི་ཚུལ་མཛད་པས་དེ་ལ་སོགས་པ་ལའོ།  ། ཁྱད་པར་དུ་ཕར་ཕྱིན་ཐ་མ་གཉིས་ལ་བསླབ་པ་ནི། ཞི་གནས་དང་ལྷག་མཐོང་བསྒོམ་པའི་ཚུལ་ཏེ། དེ་གཉིས་རིམ་པ་ལྟར་བསམ་གཏན་དང་ཤེས་རབ་ཀྱི་ཕར་ཕྱིན་གྱིས་བསྡུས་པའི་ཕྱིར་རོ། །འདི་ལ་དྲུག །ཞི་གནས་དང་ལྷག་མཐོང་བསྒོམས་པའི་ཕན་ཡོན།  དེ་གཉིས་ཏིང་ངེ་འཛིན་ཀུན་གྱི་སྡུད་བྱེད་དུ་བསྟན་པ། ཞི་གནས་དང་ལྷག་མཐོང་གི་ངོ་བོ། གཉིས་ཀ་བསྒོམ་དགོས་པའི་རྒྱུ་མཚན། གོ་རིམ་ངེས་པའི་ཚུལ། སོ་སོའི་བསླབ་ཚུལ་ལོ། །  དང་པོ་ལ་ཐེག་པ་ཆེ་ཆུང་གི་ཡོན་ཏན་ཐམས་ཅད་ཞི་ལྷག་གི་འབྲས་བུར་བསྟན་པ།  དེ་ལ་དོགས་པ་བསལ་ནས་ཕྱོགས་འདྲའི་སྒོ་ནས་དེའི་ཕན་ཡོན་དུ་བསྟན་པ།  ཐེག་པ་གཉིས་ཀྱི་ཡོན་ཏན་ཐམས་ཅད་བསྒྲུབ་པའི་ཕན་ཡོན།  སྤང་བྱ་སྤོང་བའི་ཕན་ཡོན།  དེས་མཚོན་ནས་ཕན་ཡོན་གཞན་ཡང་ཤེས་པར་བྱ་བ་དང་ལྔ་ལས།   དང་པོ་ནི། ཐེག་པ་ཆེ་ཆུང་གི་ཡོན་ཏན་འཇིག་རྟེན་པ་དང༌། འཇིག་རྟེན་ལས་འདས་པ་ཐམས་ཅད་ཞི་གནས་དང་ལྷག་མཐོང་གི་འབྲས་བུ་ཡིན་ཏེ། འཕགས་པ་དགོངས་པ་ངེས་འགྲེལ་ལས། བྱམས་པ། ཡང་ཉན་ཐོས་རྣམས་ཀྱིའམ། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིའམ། དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་དགེ་བའི་ཆོས་འཇིག་རྟེན་པ་དང་འཇིག་རྟེན་ལས་འདས་པ་ཐམས་ཅད་ཀྱང་ཞི་གནས་དང་ལྷག་མཐོང་གི་འབྲས་བུ་ཡིན་པར་རིག་པར་བྱའོ། །ཞེས་གསུངས་པའི་ཕྱིར་རོ། །  ངགཉིས་པ་དེ་ལ་དོགས་པ་དང་།  ལན་གཉིས་ལས།  དང་པོ་ནི། ཞི་གནས་དང་ལྷག་མཐོང་ནི་བསྒོམས་བྱུང་ཐོབ་པའི་རྒྱུད་ཀྱི་ཡོན་ཏན་མ་ཡིན་ནམ། ཡོན་ཏན་དེ་དག་ཐམས་ཅད་དེ་གཉིས་ཀྱི་འབྲས་བུར་ཇི་ལྟར་འཐད་སྙམ་ན།   ངགཉིས་པ་ལ་ལན་དངོས་བཤད་པ་དང་།  དེ་ཉིད་མདོ་སྡེའི་ལུང་དང་སྦྱར་བའོ།  །དང་པོནི།  ཞི་གནས་དང་ལྷག་མཐོང་དངོས་ནི་འཆད་པར་འགྱུར་བ་ལྟར་བསྒོམས་བྱུང་ཐོབ་པའི་རྒྱུད་ཀྱི་ཡོན་ཏན་ཡིན་པས་ཐེག་པ་ཆེ་ཆུང་གི་ཡོན་ཏན་ཐམས་ཅད་དེ་གཉིས་ཀྱི་འབྲས་བུ་མ་ཡིན་ཀྱང༌། དགེ་བའི་དམིགས་པ་ལ་སེམས་རྩེ་གཅིག་པ་ཡན་ཆད་ཀྱི་ཏིང་ངེ་འཛིན་རྣམས་ཞི་གནས་ཀྱི་ཕྱོགས་སུ་འདུ་ལ། ཇི་ལྟ་བའམ་ཇི་སྙེད་པའི་དོན་སོ་སོར་འབྱེད་པའི་ཤེས་རབ་དགེ་བ་རྣམས་ལྷག་མཐོང་གི་ཕྱོགས་སུ་འདུ་བས། དེ་ལ་དགོངས་ནས་ཐེག་པ་གསུམ་གྱི་ཡོན་ཏན་ཐམས་ཅད་ཞི་ལྷག་གི་འབྲས་བུར་གསུངས་པས་འགལ་བ་མེད་དོ། །  ངགཉིས་པ་ལུང་དང་སྦྱར་བ་ནི།  དེ་ལ་དགོངས་ནས་ཐེག་པ་ཆེན་པོ་ལ་དད་པ་བསྒོམ་པའི་མདོ་ལས་ཀྱང༌། རིགས་ཀྱི་བུ་རྣམ་གྲངས་འདིས་ཀྱང་འདི་ལྟར་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཐེག་པ་ཆེན་པོ་ལ་དད་པ་  བསྟེ་བསམ་པ་ཕུན་ཚོགས་དེ་དང་།  ཐེག་པ་ཆེན་པོ་ལས་འབྱུང་བ་  བསྟེ་སྦྱོར་བ་ཕུན་ཚོགས།  གང་ཅི་ཡང་རུང༌། དེ་ཐམས་ཅད་ནི་རྣམ་པར་མ་གཡེངས་པའི་སེམས་ཀྱིས་དོན་དང་ཆོས་ཡང་དག་པར་བསམས་པ་ལས་བྱུང་བར་རིག་པར་བྱའོ། །ཞེས་གསུངས་ཏེ། མ་གཡེངས་པའི་སེམས་ནི་ཞི་གནས་ཀྱི་ཕྱོགས་སེམས་རྩེ་གཅིག་པ་ཡིན་ལ། དོན་དང་ཆོས་ཡང་དག་པར་སེམས་པ་ནི་ལྷག་མཐོང་གི་ཕྱོགས་སོ་སོར་རྟོག་པའི་ཤེས་རབ་བོ། །ངགསུམ་པ་ཐེག་པ་ཆེ་ཆུང་གཉིས་ཀྱི་ཡོན་ཏན་ཐམས་ཅད་བསྒྲུབ་པའི་ཕན་ཡོན་རིགས་པས་ཀྱང་བསྒྲུབ་པ་ནི།  དེས་ན་ཐེག་པ་གཉིས་ཀྱི་ཡོན་ཏན་ཐམས་ཅད་སོ་སོར་རྟོག་པའི་ཤེས་རབ་ཀྱིས་དཔྱད་ནས་སྐྱོང་བ་དང༌། དམིགས་པ་ལ་སེམས་རྩེ་གཅིག་པ་གཉིས་ཀྱིས་བསྒྲུབ་དགོས་པ་ཡིན་གྱི། འཇོག་སྒོམ་དང་དཔྱད་སྒོམ་ཕྱོགས་རེ་བས་བསྒྲུབ་པ་མིན་ནོ། ། ངབཞི་པ་ལ་སྤང་བྱ་སྤོང་བའི་ཕན་ཡོན་དགོངས་འགྲེལ་གྱི་ལུང་དང་སྦྱར་བ་དང་།  དེའི་ལུང་དོན་བཤད་པའོ།  །དང་པོ་ནི།  ཡང་དགོངས་འགྲེལ་ལས། སྐྱེ་བ་པོ་ཡིས་  བཏེ་སྐྱེ་བོ་གང་གིས་  ལྷག་མཐོང་  བགོམས་པར་༡བྱས་པ་ དང༌། །ཞི་ག༣ནས་གོམས་པར་བྱས་ན་ནི། །ག༢ནས་ངན་ལེན་གྱི་འཆིང་བ་  བཉོན་མོངས་བསྐྱེད་པའིནུས་པ་སེམས་རྒྱུད་ལ་གནས་བའི་བག་ཆགས་ཏེ་ས་བོན་ལས་རྣམ་པར་གྲོལ་བར་འགྱུར་བ་  དང༌། །མཚ༤ན་མའི་འཆིང་  བབ་སྟེ་ཉོན་མོངས་ཀྱི་ས་བོན་ལ་གསོས་འདེབས་བྱེད་དམ་སྣོན་པ་རྒྱག་མཁན་ཉོན་མོངས་མངོན་གྱུར་བ་  ལས་རྣམ་  བཔར་  གྲོལ་  བབར་  འགྱུར་  བརོ།  །མདོར་ན་ལྷག་མཐོང་གིས་ཉོན་མོངས་ཀྱི་ས་བོན་སྤོང་།  ཞི་གནས་ཀྱིས་མངོན་གྱུར་བ་སྤོང་ཞེས་པའོ།  །  ཞེས་གསུངས་སོ། །  ངགཉིས་པ་ལུང་དོན་བཤད་པ་ནི།   དེ་ལ་གནས་ངན་ལེན་ནི། ཡུལ་ཅན་ཕྱིན་ཅི་ལོག་གོང་ནས་གོང་དུ་བསྐྱེད་པའི་ནུས་པ་སེམས་རྒྱུད་ལ་གནས་པའི་བག་ཆགས་དང་། མཚན་མ་ནི། ཡུལ་ཕྱིན་ཅི་ལོག་ལ་མངོན་པར་ཞེན་པ་སྔ་ཕྱིར་འབྱུང་བ་བག་ཆགས་དེ་གསོ་བ་དང༌། སྔ་མ་ལྷག་མཐོང་གིས་སྤོང་ལ། ཕྱི་མ་ཞི་གནས་ཀྱིས་སྤོང་བར་ཤེར་ཕྱིན་མན་ངག་ཏུ་བཤད་དོ། །དེ་དག་ནི་ཞི་ལྷག་གི་ཐ་སྙད་བཏགས་པའི་ཕན་ཡོན་ཡིན་གྱི།  ངལྔ་པ་དེས་མཚོན་ནས་ཕན་ཡོན་གཞན་ཡང་ཤེས་པར་བྱ་བ་ནི།   ཞི་ལྷག་གི་ཐ་སྙད་མ་བཏགས་ཀྱང་དོན་འདྲ་བ་བསམ་གཏན་དང་ཤེས་རབ་ཀྱི་ཕན་ཡོན་གསུངས་པ་རྣམས་ཀྱང་འདི་གཉིས་ཀྱི་ཕན་ཡོན་དུ་ཤེས་པར་བྱའོ།།          གཉིས་པ་   ངདེ་གཉིས་ཏིང་ངེ་འཛིན་ཀུན་གྱི་སྡུད་བྱེད་དུ་བསྟན་པ་ལ་གཉིས།  དཔེའི་སྒོ་ནས་ཐེག་པ་ཆེ་ཆུང་གི་ཡོན་ཏན་ཐམས་ཅད་ཀྱི་རྩ་བ་ཡིན་པས་དེ་གཉིས་སུ་འདུ་བར་བསྟན་པ་དང་། དེས་ན་བསམ་གཏན་པ་རྣམས་ཀྱིས་ཏིང་ངེ་འཛིན་གྱི་སྤྱི་ཆིངས་གཉིས་རྩད་གཅོད་དགོས་པར་བསྟན་པའོ།  །དང་པོ་ལ་གཉིས།  དཔེ་འགོད་ཅིང་དོན་བཤད་པ་དང་།  དེཉིད་ལུང་དང་སྦྱར་བའོ།  །དང་པོ་ལ་དཔེ་དོན་གཉིས་ལས།  དང་པོ་  ནི། དཔེར་ན་ཤིང་སྡོང་པོ་ལ་ཡལ་ག་དང་འདབ་མ་དང་མེ་ཏོག་དང་འབྲས་བུ་མཐའ་ཡས་པ་ཞིག་ཡོད་ཀྱང༌། དེ་དག་གང་ལ་འདུས་པའི་གནད་ནི་རྩ་བ་ཡིན་པ་ལྟར།  ངགཉིས་པ་དོན་བཤད་པ་ནི།  ཐེག་པ་ཆེ་ཆུང་གི་ཏིང་ངེ་འཛིན་མཐའ་ཡས་པ་ཞིག་བཀའ་སྩལ་པ་ཐམས་ཅད་ཀྱང༌། གང་དུ་འདུས་པའི་གནད་དམ་པ་ནི་ཞི་གནས་དང་ལྷག་མཐོང་ཡིན་ཏེ།  ངགཉིས་པ་དེ་ཉིད་ལུང་དང་སྦྱར་བ་ནི།   དགོངས་འགྲེལ་ལས། ངས་ཉན་ཐོས་རྣམས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་དང་དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་ཏིང་ངེ་འཛིན་རྣམ་པ་དུ་མ་བསྟན་པ་གང་ཡིན་པ་དེ་དག་ཐམས་ཅད་ཞི་གནས་དང་ལྷག་མཐོང་གིས་བསྡུས་པར་རིག་པར་བྱའོ། །ཞེས་གསུངས་སོ། །  ངགཉིས་པ་དེས་ན་བསམ་གཏན་པ་རྣམས་ཀྱིས་ཏིང་ངེ་འཛིན་གྱི་སྤྱི་ཆིངས་གཉིས་རྩད་གཅོད་དགོས་པར་བསྟན་པ་དང་།  དེ་གཉིས་སྒོམ་རིམ་གཉིས་ཀྱི་ལུང་དང་སྦྱར་བའོ།  །དང་པོ་ནི།  དེས་ན་ཏིང་ངེ་འཛིན་དོན་དུ་གཉེར་བ་རྣམས་ཀྱིས་གསལ་བའི་དབྱེ་བ་མཐའ་ཡས་པ་རྩད་བཅད་པར་མི་ནུས་པས་ཏིང་ངེ་འཛིན་ཀུན་གྱི་སྤྱི་ཆིངས་ཞི་ལྷག་གཉིས་ཀྱི་སྐྱོང་ཚུལ་ལེགས་པར་རྩད་བཅད་ལ་དུས་ཀུན་ཏུ་བསྟེན་པར་བྱ་སྟེ།  ངགཉིས་པ་སྒོམ་རིམ་གཉིས་ཀྱི་ལུང་དང་སྦྱར་བ་ནི།  སྒོམ་རིམ་ཐ་མ་ལས། དེ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཏིང་ངེ་འཛིན་ཐ་དད་པ་ཚད་མེད་པ་དཔག་ཏུ་མེད་པ་ལ་སོགས་པ་བསྟན་དུ་ཟིན་ཀྱང༌། ཞི་གནས་དང་ལྷག་མཐོང་གཉིས་ཀྱིས་ཏིང་ངེ་འཛིན་ཐམས་ཅད་ལ་ཁྱབ་པས་ན། ཞི་གནས་དང་ལྷག་མཐོང་ཟུང་དུ་འབྲེལ་བར་འཇུག་པའི་ལམ་དེ་ཉིད་བརྗོད་པར་བྱའོ། །   ངམདོ་ལས་ཏིང་ངེ་འཛིན་གྱི་མིང་གི་རྣམ་གྲངས་མི་འདྲ་བ་སྣ་ཚོགས་པ་གསུངས་པ་ཐམས་ཅད་ཀྱང་ཞི་ལྷག་གཉིས་པོ་འདི་ཉིད་ཁྱད་གཞིར་བྱས་ནས་དེའི་ཁྱད་པར་གྱི་དབྱེ་བ་ཡིན་པས་འདིར་ཞི་ལྷག་གཉིས་པོ་ཉིད་འཆད་ཅེས་པའོ།  །  ཞེས་པ་དང༌། སྒོམ་རིམ་བར་པ་ལས། དེ་གཉིས་ཀྱིས་ཏིང་ངེ་འཛིན་ཐམས་ཅད་བསྡུས་པའི་ཕྱིར། རྣལ་འབྱོར་པ་ཐམས་ཅད་ཀྱིས་དུས་ཐམས་ཅད་དུ་ངེས་པར་ཞི་གནས་དང་ལྷག་མཐོང་བསྟེན་པར་བྱ་སྟེ།  ཞེས་གསུངས་པ་ལྟར་རོ།།          གསུམ་པ་ལ། ཞི་གནས་ཀྱི་ངོ་བོ་ནི། དགོངས་འགྲེལ་ལས།   བཞི་གནས་བསྒོམ་པ་པོ་  དེ་གཅིག་པུ་  དར་གནས་  དབེན་པར་   བརང་  འདུག་སྟེ།   བསེམས་  ནང་དུ་ཡང་དག་   བཔར་ཏེ་མཉམ་པར་  བཞག་ནས་   བདམིགས་པ་སྔར་  ཇི་ལྟར་    བཐོས་པའི་དོན་  ལེགས་པར་བསམས་པའི་ཆོས་   བགང་ཡང་རུང་བ་   དེ་དག་ཉིད་ཡིད་ལ་བདམིགས་ནས་དེ་ཉིད་   བྱེད་ཅིང༌། སེམས་གང་གིས་ཡིད་ལ་བྱེད་པའི་སེམས་དེ་ནང་དུ་   བབྱིང་རྒོད་ཀྱི་དབང་དུ་མ་སོང་བར་    རྒྱུན་ཆགས་སུ་ཡིད་ལ་བྱེད་པས་ཡིད་ལ་བྱེད་དོ། །དེ་ལྟར་ཞུགས་ཤིང་དེ་ལ་ལན་མང་དུ་གནས་པ་   བས་   དེ་ལ་   བམཐར་   ལུས་ཤིན་ཏུ་སྦྱངས་པ་དང་སེམས་ཤིན་ཏུ་སྦྱངས་པ་འབྱུང་   བཞིང་དེ་ལྟར་འབྱུང་ བ་གང་ཡིན་པ་   བའི་ཏིང་ངེ་འཛིན།   དེ་ནི་ཞི་གནས་   བཡིན་ནོ་  ཞེས་བྱ་སྟེ། དེ་ལྟར་ན་བྱང་ཆུབ་སེམས་དཔའ་ཞི་གནས་ཡོངས་སུ་ཚོལ་བར་བྱེད་པ་ཡིན་ནོ   ངཞི་གནས་ཀྱི་ངོ་བོ་སྟོན་པའི་དགོངས་འགྲེལ་གྱི་མདོ་འདིས་ཞི་གནས་གྲུབ་པའི་ཚད་དང་བཅས་པ་ཡོངས་སུ་རྫོགས་པར་བསྟན་ཀྱང་ལྷག་མཐོང་གི་ངོ་བོ་སྟོན་པའི་མདོ་ལས་ལྷག་མཐོང་སྒྲུབ་ཚུལ་བསྟན་ཀྱང་གྲུབ་པའི་ཚད་ཡོངས་སུ་རྫོགས་པར་དངོས་སུ་མ་བསྟན་ནོ།  །   ཞེས་གསུངས་པ་ལྟར་རོ། །དེའི་དོན་གསུང་རབ་ཡན་ལག་བཅུ་གཉིས་ཀྱི་དོན་གང་ངེས་པར་བྱས་པ་ཕུང་པོ་ལ་སོགས་པའི་དོན་གང་ཡང་རུང་བ་ཞིག་དམིགས་ཡུལ་དུ་བྱས་ནས། དེ་ལ་དམིགས་པའི་སེམས་དེ་གཞན་དུ་མི་གཡེང་བར་དྲན་ཤེས་ཀྱིས་རྒྱུན་ཆགས་སུ་དམིགས་པ་དེ་ལ་བཏགས་པས་སེམས་དམིགས་པ་ལ་རང་གི་ངང་གིས་གནས་ཏེ། ལུས་སེམས་ཤིན་ཏུ་སྦྱངས་པའི་དགའ་བདེ་སྐྱེས་པ་ན་ཏིང་ངེ་འཛིན་དེ་ཞི་གནས་སུ་འགྱུར་ཏེ། འདི་ནི་སེམས་དམིགས་པ་ལས་མི་གཡེང་བར་ནང་དུ་བཟུང་བ་ཙམ་ལས་སྐྱེ་བར་འགྱུར་གྱི་དངོས་པོའི་དེ་ཁོ་ན་ཉིད་ཁོང་དུ་ཆུད་པ་ལ་མི་ལྟོས་སོ།།
